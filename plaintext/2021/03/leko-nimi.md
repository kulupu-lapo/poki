leko nimi toki pona

tan jan Ali

# soweli

soweli li pali mute\
ona li wile ala\
taso ona li pali\
soweli li wile pali

# pu ala

pu li suli ala\
tenpo suno kama la\
ona li moli tan\
ni ale li moli

# kasi lili

telo sewi li lon\
li tawa kasi lili\
kasi supa li telo\
ona li pilin pona

# waso

sina o kama waso\
o waso tawa ma\
ma pi jan pona\
ma pi olin mute
